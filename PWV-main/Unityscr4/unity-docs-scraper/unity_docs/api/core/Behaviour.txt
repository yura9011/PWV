// Unity Class: Behaviour
// Documentation: https://docs.unity3d.com/ScriptReference/Behaviour.html

// Description: Behaviours are Components that can be enabled or disabled.

// Properties:
// - enabled: Enabled Behaviours are Updated, disabled Behaviours are not.
// - isActiveAndEnabled: Checks whether a component is enabled, attached to a GameObject that is active in the hierarchy, and the component's OnEnable has been called.

// Methods:
// - BroadcastMessage(): Calls the method named methodName on every MonoBehaviour in this game object or any of its children.
// - CompareTag(): Checks the GameObject's tag against the defined tag.
// - GetComponent(): Gets a reference to a component of type T on the same GameObject as the component specified.
// - GetComponentInChildren(): Gets a reference to a component of type T on the same GameObject as the component specified, or any child of the GameObject.
// - GetComponentIndex(): Gets the index of the component on its parent GameObject.
// - GetComponentInParent(): Gets a reference to a component of type T on the same GameObject as the component specified, or any parent of the GameObject.
// - GetComponents(): Gets references to all components of type T on the same GameObject as the component specified.
// - GetComponentsInChildren(): Gets references to all components of type T on the same GameObject as the component specified, and any child of the GameObject.
// - GetComponentsInParent(): Gets references to all components of type T on the same GameObject as the component specified, and any parent of the GameObject.
// - SendMessage(): Calls the method named methodName on every MonoBehaviour in this game object.
// - SendMessageUpwards(): Calls the method named methodName on every MonoBehaviour in this game object and on every ancestor of the behaviour.
// - TryGetComponent(): Gets the component of the specified type, if it exists.
// - GetInstanceID(): Gets the instance ID of the object.
// - ToString(): Returns the name of the object.

// Pseudo-code representation of Behaviour
public class Behaviour {
    public var enabled; // Enabled Behaviours are Updated, disabled Behaviours are not.
    public var isActiveAndEnabled; // Checks whether a component is enabled, attached to a GameObject that is active in the hierarchy, and the component's OnEnable has been called.
    public void BroadcastMessage(); // Calls the method named methodName on every MonoBehaviour in this game object or any of its children.
    public void CompareTag(); // Checks the GameObject's tag against the defined tag.
    public void GetComponent(); // Gets a reference to a component of type T on the same GameObject as the component specified.
    public void GetComponentInChildren(); // Gets a reference to a component of type T on the same GameObject as the component specified, or any child of the GameObject.
    public void GetComponentIndex(); // Gets the index of the component on its parent GameObject.
    public void GetComponentInParent(); // Gets a reference to a component of type T on the same GameObject as the component specified, or any parent of the GameObject.
    public void GetComponents(); // Gets references to all components of type T on the same GameObject as the component specified.
    public void GetComponentsInChildren(); // Gets references to all components of type T on the same GameObject as the component specified, and any child of the GameObject.
    public void GetComponentsInParent(); // Gets references to all components of type T on the same GameObject as the component specified, and any parent of the GameObject.
    public void SendMessage(); // Calls the method named methodName on every MonoBehaviour in this game object.
    public void SendMessageUpwards(); // Calls the method named methodName on every MonoBehaviour in this game object and on every ancestor of the behaviour.
    public void TryGetComponent(); // Gets the component of the specified type, if it exists.
    public void GetInstanceID(); // Gets the instance ID of the object.
    public void ToString(); // Returns the name of the object.
}
