// Unity Class: AccessibilityHierarchy
// Documentation: https://docs.unity3d.com/ScriptReference/Accessibility.AccessibilityHierarchy.html

// Description: The hierarchy data model that the screen reader uses to navigate and interact with a Unity application.

// Properties:
// - rootNodes: The root nodes of the hierarchy.

// Methods:
// - AddNode: Creates a new AccessibilityNode with the given label and adds it to the hierarchy under the given pa
// - Clear: Resets the hierarchy to an empty state, removing all nodes and the screen reader focus.
// - ContainsNode: Verifies whether the given node exists in the accessibility hierarchy.
// - GetLowestCommonAncestor: Retrieves the lowest common ancestor of two nodes in the accessibility hierarchy.
// - InsertNode: Creates a new AccessibilityNode with the given label and inserts it at the given index in the hierar
// - MoveNode: Moves the node elsewhere in the accessibility hierarchy. For example, under a different parent or at
// - RefreshNodeFrames: Refreshes the AccessibilityNode.frame of all nodes in the accessibility hierarchy.
// - RemoveNode: Removes the node from the accessibility hierarchy and removes or re-parents its descendants.
// - TryGetNode: Tries to retrieve the AccessibilityNode with the given ID in the hierarchy.
// - TryGetNodeAt: Tries to retrieve the AccessibilityNode at the given screen coordinates.

public class AccessibilityHierarchy {
    public var rootNodes;
    public void AddNode();
    public void Clear();
    public void ContainsNode();
    public void GetLowestCommonAncestor();
    public void InsertNode();
    public void MoveNode();
    public void RefreshNodeFrames();
    public void RemoveNode();
    public void TryGetNode();
    public void TryGetNodeAt();
}
