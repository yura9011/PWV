// Pseudo-code representation of Bounds
public class Bounds {
    public var center; // The center of the bounding box.
    public var extents; // The extents of the Bounding Box. This is always half of the size of the Bounds.
    public var max; // The maximal point of the box. This is always equal to center+extents.
    public var min; // The minimal point of the box. This is always equal to center-extents.
    public var size; // The total size of the box. This is always twice as large as the extents.
    public void ClosestPoint(); // The closest point on the bounding box.
    public void Contains(); // Is point contained in the bounding box?
    public void Encapsulate(); // Grows the Bounds to include the point.
    public void Expand(); // Expand the bounds by: size + (amount * 2) along each side.
    public void IntersectRay(); // Does ray intersect this bounding box?
    public void Intersects(); // Does another bounding box intersect with this bounding box?
    public void SetMinMax(); // Sets the bounds to the min and max value of the box.
    public void SqrDistance(); // The smallest squared distance between the point and this bounding box.
    public void ToString(); // Returns a formatted string for the bounds.
}
