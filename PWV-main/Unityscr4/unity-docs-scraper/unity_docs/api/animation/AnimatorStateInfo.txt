// Pseudo-code representation of AnimatorStateInfo
public class AnimatorStateInfo {
    public var fullPathHash; // The full path hash for this state.
    public var length; // Current duration of the state.
    public var loop; // Is the state looping.
    public var normalizedTime; // Normalized time of the State.
    public var shortNameHash; // The hash is generated using Animator.StringToHash. The hash does not include the name of the parent layer.
    public var speed; // The playback speed of the animation. 1 is the normal playback speed.
    public var speedMultiplier; // The speed multiplier for this state.
    public var tagHash; // The Tag of the State.
    public void IsName(); // Checks if name matches the name of the active state in the state machine.
    public void IsTag(); // Checks whether the animation state has the specified tag.
}
