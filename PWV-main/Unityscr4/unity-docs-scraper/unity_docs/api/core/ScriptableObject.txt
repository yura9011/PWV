public class ScriptableObject {
    public var hideFlags; // Should the object be hidden, saved with the Scene or modifiable by the user?
    public var name; // The name of the object.
    public void GetInstanceID(); // Gets the instance ID of the object.
    public void ToString(); // Returns the name of the object.
    public void CreateInstance(); // Creates an instance of a scriptable object.
}
