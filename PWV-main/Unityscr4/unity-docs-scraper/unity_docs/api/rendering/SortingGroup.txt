// Unity Class: SortingGroup
// Documentation: https://docs.unity3d.com/ScriptReference/Rendering.SortingGroup.html

// Description: Adding a SortingGroup component to a GameObject will ensure that all Renderers within the GameObject's descendants will be sorted and rendered together.

// Properties:
// - sortAtRoot: Ignores any parent SortingGroup and sorts this and its descendant Renderers against other Renderers 
// - sortingLayerID: Unique ID of the Renderer's sorting layer.
// - sortingLayerName: Name of the Renderer's sorting layer.
// - sortingOrder: Renderer's order within a sorting layer.

// Methods:
// - BroadcastMessage: Calls the method named methodName on every MonoBehaviour in this game object or any of its children.
// - CompareTag: Checks the GameObject's tag against the defined tag.
// - GetComponent: Gets a reference to a component of type T on the same GameObject as the component specified.
// - GetComponentInChildren: Gets a reference to a component of type T on the same GameObject as the component specified, or any 
// - GetComponentIndex: Gets the index of the component on its parent GameObject.
// - GetComponentInParent: Gets a reference to a component of type T on the same GameObject as the component specified, or any 
// - GetComponents: Gets references to all components of type T on the same GameObject as the component specified.
// - GetComponentsInChildren: Gets references to all components of type T on the same GameObject as the component specified, and a
// - GetComponentsInParent: Gets references to all components of type T on the same GameObject as the component specified, and a
// - SendMessage: Calls the method named methodName on every MonoBehaviour in this game object.
// - SendMessageUpwards: Calls the method named methodName on every MonoBehaviour in this game object and on every ancestor o
// - TryGetComponent: Gets the component of the specified type, if it exists.
// - GetInstanceID: Gets the instance ID of the object.
// - ToString: Returns the name of the object.
// - UpdateAllSortingGroups: Updates all Sorting Group immediately.

public class SortingGroup {
    public var sortAtRoot;
    public var sortingLayerID;
    public var sortingLayerName;
    public var sortingOrder;
    public void BroadcastMessage();
    public void CompareTag();
    public void GetComponent();
    public void GetComponentInChildren();
    public void GetComponentIndex();
    public void GetComponentInParent();
    public void GetComponents();
    public void GetComponentsInChildren();
    public void GetComponentsInParent();
    public void SendMessage();
    public void SendMessageUpwards();
    public void TryGetComponent();
    public void GetInstanceID();
    public void ToString();
    public void UpdateAllSortingGroups();
}
