// Unity Class: GameObject
// Documentation: https://docs.unity3d.com/ScriptReference/GameObject.html

// Description: Base class for all objects that can exist in a scene. Add components to a GameObject to control its appearance and behavior.

// Properties:
// - activeInHierarchy: The active state of the GameObject in the Scene hierarchy. True if active, false if inactive. (Read Only)
// - activeSelf: The local active state of the GameObject. True if active, false if inactive. (Read Only)
// - isStatic: Whether there are any Static Editor Flags set for the GameObject.
// - layer: Integer identifying the layer the GameObject is assigned to.
// - scene: The Scene that contains the GameObject.
// - sceneCullingMask: The Scene culling mask defined for the GameObject. (Read Only)
// - tag: The tag assigned to the GameObject.
// - transform: The Transform attached to the GameObject (Read Only).
// - transformHandle: The TransformHandle of the GameObject (Read Only).

// Constructors:
// - GameObject(): Creates a new GameObject, with optional parameters to specify a name and set of components to attach.

// Methods:
// - AddComponent(): Adds a component of the specified type to the GameObject.
// - BroadcastMessage(): Calls the specified method on every MonoBehaviour attached to the GameObject or any of its children.
// - CompareTag(): Checks if the specified tag is attached to the GameObject.
// - GetComponent(): Retrieves a reference to a component of the specified type, by providing the component type as a type parameter to the generic method.
// - GetComponentAtIndex(): Retrieves a reference to a component at a specified index of the GameObject's array of components.
// - GetComponentCount(): Retrieves the total number of components currently attached to the GameObject.
// - GetComponentInChildren(): Retrieves a reference to a component of type T on the specified GameObject, or any child of the GameObject.
// - GetComponentIndex(): Retrieves the index of the specified component in the array of components attached to the GameObject.
// - GetComponentInParent(): Retrieves a reference to a component of type T on the specified GameObject, or any parent of the GameObject.
// - GetComponents(): Retrieves references to all components of type T on the specified GameObject.
// - GetComponentsInChildren(): Retrieves references to all components of type T on the specified GameObject, and any child of the GameObject.
// - GetComponentsInParent(): Retrieves references to all components of type T on the specified GameObject, and any parent of the GameObject.
// - SendMessage(): Calls the specified method on every MonoBehaviour attached to the GameObject.
// - SendMessageUpwards(): Calls the specified method on every MonoBehaviour attached to the GameObject and on every ancestor of the behaviour.
// - SetActive(): Activates or deactivates the GameObject locally, according to the value of the supplied parameter.
// - TryGetComponent(): Retrieves the component of the specified type, if it exists.

// Example 1:
//   using UnityEngine;  public class Example_GameObject : MonoBehaviour
//     {

//         }

// Pseudo-code representation of GameObject
public class GameObject {
    public var activeInHierarchy; // The active state of the GameObject in the Scene hierarchy. True if active, false if inactive. (Read Only)
    public var activeSelf; // The local active state of the GameObject. True if active, false if inactive. (Read Only)
    public var isStatic; // Whether there are any Static Editor Flags set for the GameObject.
    public var layer; // Integer identifying the layer the GameObject is assigned to.
    public var scene; // The Scene that contains the GameObject.
    public var sceneCullingMask; // The Scene culling mask defined for the GameObject. (Read Only)
    public var tag; // The tag assigned to the GameObject.
    public var transform; // The Transform attached to the GameObject (Read Only).
    public var transformHandle; // The TransformHandle of the GameObject (Read Only).
    public void AddComponent(); // Adds a component of the specified type to the GameObject.
    public void BroadcastMessage(); // Calls the specified method on every MonoBehaviour attached to the GameObject or any of its children.
    public void CompareTag(); // Checks if the specified tag is attached to the GameObject.
    public void GetComponent(); // Retrieves a reference to a component of the specified type, by providing the component type as a type parameter to the generic method.
    public void GetComponentAtIndex(); // Retrieves a reference to a component at a specified index of the GameObject's array of components.
    public void GetComponentCount(); // Retrieves the total number of components currently attached to the GameObject.
    public void GetComponentInChildren(); // Retrieves a reference to a component of type T on the specified GameObject, or any child of the GameObject.
    public void GetComponentIndex(); // Retrieves the index of the specified component in the array of components attached to the GameObject.
    public void GetComponentInParent(); // Retrieves a reference to a component of type T on the specified GameObject, or any parent of the GameObject.
    public void GetComponents(); // Retrieves references to all components of type T on the specified GameObject.
    public void GetComponentsInChildren(); // Retrieves references to all components of type T on the specified GameObject, and any child of the GameObject.
    public void GetComponentsInParent(); // Retrieves references to all components of type T on the specified GameObject, and any parent of the GameObject.
    public void SendMessage(); // Calls the specified method on every MonoBehaviour attached to the GameObject.
    public void SendMessageUpwards(); // Calls the specified method on every MonoBehaviour attached to the GameObject and on every ancestor of the behaviour.
    public void SetActive(); // Activates or deactivates the GameObject locally, according to the value of the supplied parameter.
    public void TryGetComponent(); // Retrieves the component of the specified type, if it exists.
}
